Variable A: Type.

all/some

Fixpoint all P (l: list A): Prop := match l with
| nil ⇒ True
| cons b bs ⇒ P b ∧ all P bs
end.

Fixpoint allT P (l: list A): Type := match l with
| nil ⇒ True : Type
| cons b bs ⇒ prod (P b) (allT P bs)
end.

Lemma all_PROP: ∀ {P l},
  all (fun x ⇒ ☐ P x) l
  → ☐ allT (fun x ⇒ P x) l.

Lemma allT_all: ∀ (P: A → Prop) l, allT P l → all P l.

Fixpoint some P (l: list A) := match l with
| nil ⇒ False
| cons b bs ⇒ P b ∨ some P bs
end.

Fixpoint someT P (l: list A) : Type := match l with
| nil ⇒ False : Type
| cons b bs ⇒ sum (P b) (someT P bs)
end.

Lemma someT_some: ∀ (P: A → Prop) l, someT P l → some P l.
Lemma some_someT: ∀ (P: A → Prop) l, some P l → ☐ (someT P l).

Lemma allT_mono: ∀ (P: A → Type) (Q: A → Type),
  (∀ x, P x → Q x) →
  ∀ l, allT P l → allT Q l.

Lemma all_mono: ∀ (P Q: A → Prop),
  (∀ x, P x → Q x)
    → ∀ l, all P l → all Q l.

Lemma all_ext: ∀ (P Q: A → Prop),
  (∀ x, P x ↔ Q x)
    → ∀ l, all P l ↔ all Q l.

Lemma all_In: ∀ (P: A → Prop) l,
  all P l ↔ ∀ x, In x l → P x.

Lemma some_In: ∀ (P: A → Prop) l,
  some P l ↔ ∃ x, In x l ∧ P x.

Lemma cons_not_nil: ∀ x l, @cons A x l ≠ nil ↔ True.
