Inductive set :=
  S : ∀ A (p : @boolean A) (h: A → set) X (f: X → set), set.


Inductive lt : set → set → Prop :=
  | lt_f : ∀ A p h X f i, lt (f i) (S A p h X f)
  | lt_h : ∀ A p h X f i, lt (h i) (S A p h X f)
.
Infix "<" := lt.

Lemma wf_lt : well_founded lt.
