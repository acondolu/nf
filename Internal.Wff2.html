Replace an element of a list with another list:
Definition replace : ∀ {i: nat} {l: list A},
  i < length l → list A → list A.

Lemma replace_succ: ∀ {i l l' x} (p: S i < length (x::l)) (p': i < length l),
  replace p l' = x :: replace p' l'.

Get an element of the list by its index:
Definition get : ∀ {i: nat} {l: list A}, i < length l → A.

Lemma get_succ: ∀ {i l x} (p: S i < length (x::l)) (p': i < length (l)),
  get p = get p'.

Fixpoint all P (x: list A) := match x with
| nil ⇒ True
| cons b bs ⇒ P b ∧ all P bs
end.

Inductive lt_lst : list A → list A → Prop :=
  | C: ∀ i (l l': list A) (p: i < length l),
         all (fun x ⇒ lt x (get p)) l'
         →
         lt_lst (replace p l') l
.

Lemma l2p1: ∀ {a M0},
  (∀ b, lt b a → ∀ M, Acc lt_lst M → Acc lt_lst (b :: M))
  → Acc lt_lst M0
  → (∀ M, lt_lst M M0 → Acc lt_lst (a :: M))
  → Acc lt_lst (a :: M0).

Lemma l2p2: ∀ x,

→ ∀ M, (Acc lt_lst M) → Acc lt_lst (x :: M).

Lemma l2p3: ∀ {M} a, Acc lt_lst M → Acc lt_lst (cons a M).

Theorem wf_lst: well_founded lt_lst.

Definition permle l l' := ∃ l'', Permutation l l'' ∧ lt_lst l'' l'.

Lemma perm_lt_dx: ∀ {a b b'}, Permutation b b' → lt_lst a b' → permle a b.

Lemma perm_Acc : ∀ l l', Permutation l l' → Acc permle l → Acc permle l'.

Theorem wf_perm : well_founded permle.
