Fixpoint bool a := match a with
  | Pos f ⇒ ∃ x, ¬ bool (f x)
  | Neg f ⇒ ∀ x, bool (f x)
end.

Lemma bool_bot : ¬ bool Ø.

Lemma bool_neg : ∀ a, bool (sing a) ↔ ¬ bool a.

Lemma bool_and : ∀ a b, bool (sing (sing a ∪ sing b)) ↔ bool a ∧ bool b.

Lemma bool_or : ∀ a b, bool (sing (sing a) ∪ sing (sing b)) ↔ bool a ∨ bool b.
