Lemma low_empty: low Ø.
Proof. simpl low. auto. Qed.

Lemma low_sing : ∀ x, low (sing x).
Proof. unfold sing; simpl low; tauto. Qed.

Lemma low_neg: ∀ x, low (neg x) ↔ ¬ low x.
Proof.
  destruct x; unfold neg; simpl low; tauto.
Qed.

Lemma low_cap: ∀ x y, low (x ⋂ y) ↔ (low x ∨ low y).
Proof.
  destruct x; destruct y; simpl cap; simpl low; tauto.
Qed.

Lemma low_cup: ∀ x y, low (x ∪ y) ↔ (low x ∧ low y).
Proof.
  destruct x; destruct y; simpl cap; simpl low; tauto.
Qed.



Definition pred_sound P := ∀ x y, x ≡ y → (P x ↔ P y).
Definition compr: ∀ {x}, low x → (set → Prop) → set.
Proof.
  intros x low P.
  destruct x; destruct low.
  apply (@Low {x : X & P (s x)}).
  intro. destruct X0. exact (s x).
Defined.

Lemma compr_ok: ∀ {x} P (low: low x) (pok: pred_sound P), ∀ z, IN z (compr low P) ↔ IN z x ∧ P z.
Proof.
  intros. destruct x; destruct low.
  simpl IN. split; intros; destruct H.
  - destruct x. split. ∃ x. assumption.
    apply (pok _ _ H). assumption.
  - destruct H. destruct (pok _ _ H).
    ∃ (existT _ x (H2 H0)). assumption.
Qed.
