Definition cup B C :=
  match B, C with S X Y f g e, S X' Y' f' g' e' ⇒
  S _ _
    (select f (aux f' g e g' e' ∘ f) ⨁ select f' (aux f g' e' g e ∘ f'))
      (g ⨁ g') (Or (map inl e) (map inr e'))
end.
