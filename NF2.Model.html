Library NF2.Model


Inductive set :=
  | Pos : ∀ {X}, (X → set) → set
  | Neg : ∀ {X}, (X → set) → set
.
Notation 𝓥 := set.

Fixpoint eeq a b := match a,b with
  | Pos _, Neg _ ⇒ False
  | Neg _, Pos _ ⇒ False
  | Pos f, Pos g ⇒
      (∀ x, ∃ y, eeq (f x) (g y)) ∧ (∀ y, ∃ x, eeq (f x) (g y))
  | Neg f, Neg g ⇒
      (∀ x, ∃ y, eeq (f x) (g y)) ∧ (∀ y, ∃ x, eeq (f x) (g y))
end.
Notation "A ≡ B" := (eeq A B) (at level 85).

Definition iin a b := match b with
  | Pos f ⇒ ∃ x, eeq (f x) a
  | Neg f ⇒ ∀ x, ¬ eeq (f x) a
end.
Notation "A ∈ B" := (iin A B) (at level 85).


Lemma eeq_refl : ∀ {x}, x ≡ x.

Lemma eeq_trans : ∀ {x y z}, x ≡ y → y ≡ z → x ≡ z.

Lemma eeq_sym : ∀ {x y}, x ≡ y → y ≡ x.

Lemma in_sound_right:
  ∀ {x y}, x ≡ y → ∀ z, z ∈ x → z ∈ y.

Lemma in_sound_left:
  ∀ {x y}, x ≡ y → ∀ z, x ∈ z → y ∈ z.

Definition low x := match x with
  | Pos _ ⇒ True
  | Neg _ ⇒ False
