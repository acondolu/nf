Definition eeq' : set × set → Prop.

Definition eeq x y := eeq' (x, y).
Infix "==" := eeq (at level 50) : type_scope.

Definition Aeq {X Y} f g :=
  (∀ x: X, ∃ y, f x == g y)
  ∧ ∀ y: Y, ∃ x, f x == g y.

Lemma Aeq_refl: ∀ X f, @Aeq X X f f.

Lemma Aeq_sym: ∀ X Y f g, @Aeq X Y f g → Aeq g f.

Lemma Aeq_trans: ∀ {X Y Z} f g h, @Aeq X Y f g → Aeq g h → @Aeq X Z f h.


Definition Qeq := eeq_boolean eeq.

Lemma eeq_boolean_qeq:
  ∀ {X Y p p'} {h: X → set} {h': Y → set},
    eeq_boolean
      (eeq ⨀ (h ⨁ h'))
        (map inl p) (map inr p')
    ↔
    Qeq (map h p) (map h' p').

Lemma eeq_unfold {A p h X f A' p' h' X' f'}:
  eeq (S A p h X f) (S A' p' h' X' f')
  ↔
  Aeq f f'
    ∧
  Qeq (map h p) (map h' p')
