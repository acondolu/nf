Extensionally equivalent predicates
Definition extP {X} P Q := ∀ x: X, P x ↔ Q x.

Extensionally equivalent binary predicates
Definition extR {X Y} P Q := ∀ x: X, ∀ y: Y, P x y ↔ Q x y.

Definition respects {X} (R: X → X → Prop) (P: X → Prop) :=
  ∀ x x', R x x' → (P x ↔ P x').

Lemma respects_ext {X} (R R': X → X → Prop) :
  extR R R' → extP (respects R) (respects R').


Lemma respects_swap {X Y Z R} :
  ∀ (f: X → Z) (g: Y → Z) h,
    respects (R ⨀ (f ⨁ g)) h
      → respects (R ⨀ (g ⨁ f)) (h ∘ swap).


Lemma invert_sum_respects {X Y Z h h'} {R: Z → Z → Prop} {P: X + Y → Prop}:
  Equivalence R →
    respects (R ⨀ (h ⨁ h')) P → respects R (invert_sum P (R ∘ h) (R ∘ h')).
