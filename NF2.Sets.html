Definition 𝒰 := High False (fun x ⇒ match x with end).

Lemma univ_ok : ∀ x, x ∈ 𝒰.
Proof. intros x H. destruct H. Qed.

Definition emptyset := Low False (fun x ⇒ match x with end).
Notation Ø := emptyset.

Lemma emptyset_ok : ∀ x, ¬ (x ∈ Ø).
Proof. intros x H. apply H. Qed.

Definition compl : SET → SET :=
  fun x ⇒ match x with
  | Low _ f ⇒ High _ f
  | High _ f ⇒ Low _ f
  end
.

Lemma compl_ok : ∀ x y, x ∈ compl y ↔ (x ∈ y → False).
Proof.
  intros x y. destruct y; simpl compl; simpl IN; split; firstorder.
  apply not_all_not_ex. assumption.
Qed.

Definition sing : SET → SET :=
  fun x ⇒ Low unit (fun _ ⇒ x).

Definition sing_ok : ∀ x y, x ∈ sing y ↔ y ≡ x.
Proof.
  intros. simpl. split; intros. destruct H; firstorder. apply ex_unit. auto.
Qed.


Definition minus {X Y} f g : { x : X & ∀ y : Y, ¬ (g y ≡ f x) } → SET :=
  select f (fun x ⇒ ∀ y, ¬ (g y ≡ f x)).

Definition meet {X Y} f g : { x : X & ∃ y : Y, g y ≡ f x } → SET :=
  select f (fun x ⇒ ∃ y, g y ≡ f x).

Definition join {X Y} f g : X + Y → SET := f ⨁ g.

Definition cap x y := match x, y with
  | Low _ f, High _ g ⇒ Low _ (minus f g)
  | High _ f, Low _ g ⇒ Low _ (minus g f)
  | Low _ f, Low _ g ⇒ Low _ (meet f g)
  | High _ f, High _ g ⇒ High _ (join f g)
end.
Notation "A ⋂ B" := (cap A B) (at level 85).

Lemma cap1: ∀ x y z, z ∈ cap x y → (z ∈ x) ∧ (z ∈ y).
Proof.
  destruct x; destruct y; simpl; intro z.
  - rewrite (ex_T (fun x ⇒ ∃ y, _) (fun x ⇒ s x ≡ z)).
    firstorder. rewrite<- H in H0. firstorder.
  - rewrite (ex_T (fun x ⇒ ∀ y, ¬ _) (fun x ⇒ s x ≡ z)).
    firstorder. rewrite<- H0. apply H.
  - rewrite (ex_T (fun x ⇒ ∀ y, ¬ _) (fun x ⇒ s0 x ≡ z)).
    firstorder. setoid_rewrite H0 in H. firstorder.
  - unfold join. firstorder. apply (H (inl x)). apply (H (inr x)).
Qed.

Lemma cap2: ∀ x y z, (z ∈ x) ∧ (z ∈ y) → z ∈ cap x y.
Proof.
  destruct x; destruct y; simpl; unfold meet, minus, join, select; intros; destruct H.
  - rewrite (ex_T (fun x ⇒ ∃ y, _) (fun x ⇒ s x ≡ z)).
    destruct H, H0. ∃ x. firstorder. ∃ x0. transitivity z. auto. symmetry. auto.
  - rewrite (ex_T (fun x ⇒ ∀ y, ¬ _) (fun x ⇒ s x ≡ z)).
    destruct H. ∃ x. firstorder. simpl in H0. setoid_rewrite<- H in H0. apply (H0 y).
  - rewrite (ex_T (fun x ⇒ ∀ y, ¬ _) (fun x ⇒ s0 x ≡ z)).
    destruct H0. ∃ x. firstorder. setoid_rewrite H0.
    apply (H y).
  - destruct x. apply (H x). apply (H0 x).
Qed.

Lemma cap_ok : ∀ x y z, z ∈ (x ⋂ y) ↔ (z ∈ x) ∧ (z ∈ y).
Proof. intros. split. apply cap1. apply cap2. Qed.

Definition cup x y := compl (cap (compl x) (compl y)).
Notation "A ∪ B" := (cup A B) (at level 85).

Lemma cup_ok : ∀ x y z, z ∈ (x ∪ y) ↔ (z ∈ x) ∨ (z ∈ y).
Proof.
  intros. unfold cup. rewrite compl_ok. rewrite cap_ok. repeat rewrite compl_ok. tauto.
Qed.
