Theorem weak_regularity: ∀ {x}, low x → IN x x → False.
Proof.
  induction x; simpl low; auto; intros.
  assert (H1' := H1). destruct H1. apply (H x).
  - destruct (s x); simpl; auto.
  - setoid_rewrite H1. assumption.
Qed.


Lemma pos_univ: ∀ X f Y g,
  (∀ z, IN z (Low X f) ↔ IN z (High Y g))
  → ∀ z, IN z (Low (X + Y) (f ⨁ g)).
Proof.
  intros. simpl IN in ×.
  pose proof (H z).
  destruct (classic (∃ x : X, EQ (f x) z)).
  - destruct H1. ∃ (inl x). tauto.
  - cut (∃ y : Y, EQ (g y) z). intro. destruct H2.
    ∃ (inr x). auto.
  destruct H0. pose proof (fun T ⇒ H1 (H2 T)).
  apply NNPP. intro. apply H3. intros x HH. apply H4. ∃ x. auto.
Qed.

Lemma pos_neg_ext_neq: ∀ X f Y g,
  (∀ z, IN z (Low X f) ↔ IN z (High Y g)) → False.
Proof.
  intros.
  pose proof (pos_univ _ _ _ _ H).
  apply (@weak_regularity (Low (X + Y) (f ⨁ g)) I). apply H0.
Qed.

Lemma ext_pos: ∀ X f Y g,
  (∀ z, IN z (Low X f) ↔ IN z (Low Y g))
  → EQ (Low X f) (Low Y g).
Proof.
  intros. simpl; split; intro.
  - destruct (H (f x)). simpl IN in H0.
    cut (∃ x0 : X, EQ (f x0) (f x)). intro. destruct (H0 H2). ∃ x0. apply EQ_sym. assumption. ∃ x. apply EQ_refl.
  - destruct (H (g y)). simpl IN in H1.
  cut (∃ x : Y, EQ (g x) (g y)). intro. destruct (H1 H2). ∃ x. assumption. ∃ y. apply EQ_refl.
Qed.

Lemma not_ex_all_not_iff: ∀ {U} (P: U → Prop),
  ¬ (∃ n : U, P n) ↔ ∀ n:U, ¬ P n.
Proof.
  split. apply not_ex_all_not. firstorder.
Qed.

Lemma ext_neg: ∀ X f Y g,
  (∀ z, IN z (High X f) ↔ IN z (High Y g))
  → EQ (High X f) (High Y g).
Proof.
  simpl. intros. apply (ext_pos X f Y g). simpl. intro z. specialize H with z.
  setoid_rewrite<- not_ex_all_not_iff in H. tauto.
Qed.
